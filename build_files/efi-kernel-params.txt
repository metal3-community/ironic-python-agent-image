# EFI Kernel Parameters for Raspberry Pi UEFI Boot
# These parameters should be added to the kernel command line for proper EFI support

# Essential EFI parameters
efi=runtime                    # Enable EFI runtime services
acpi=on                        # Enable ACPI (if supported)
console=tty0                   # Primary console on display
console=ttyS0,115200           # Serial console fallback
console=ttyAMA0,115200         # ARM serial console

# Memory and initialization
earlycon=uart8250,mmio32,0xfe215040    # Early console for RPi4
earlycon=pl011,mmio32,0xfe201000       # Alternative early console

# Network and PCI
pcie_aspm=off                  # Disable PCIe ASPM for stability
iommu.passthrough=1           # IOMMU passthrough mode

# Debug options (can be removed in production)
earlyprintk                   # Early kernel printk
debug                         # Enable debug messages

# Example complete kernel command line for iPXE:
# efi=runtime console=tty0 console=ttyS0,115200 console=ttyAMA0,115200 earlycon=uart8250,mmio32,0xfe215040 ip=dhcp
